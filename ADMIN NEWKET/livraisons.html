function updateStats() { const orders = getFilteredOrders(); document.getElementById('statTotalDeliv').textContent = orders.length; document.getElementById('statPending').textContent = orders.filter(o => o.status === 'En attente').length; document.getElementById('statInTransit').textContent = orders.filter(o => o.status === 'En livraison').length; document.getElementById('statDelivered').textContent = orders.filter(o => o.status === 'Livré').length; } function exportDeliveries() { const orders = getFilteredOrders(); if (!orders.length) { showToast('Aucune livraison à exporter.', 'info'); return; } const headers = ['ID', 'Client', 'Email', 'Téléphone', 'Articles', 'Date', 'Total', 'Statut']; const rows = orders.map(o => [o.id, o.customer?.name || 'Client', o.customer?.email || '', o.phone_number || '', (o.items || []).length, new Date(o.date).toLocaleDateString('fr-FR'), o.total, o.status]); const csv = [headers.join(','), ...rows.map(r => r.join(','))].join('\n'); const link = document.createElement('a'); link.setAttribute('href', URL.createObjectURL(new Blob([csv], { type: 'text/csv;charset=utf-8;' }))); link.setAttribute('download', `newket-livraisons-${new Date().toISOString().split('T')[0]}.csv`); link.style.visibility = 'hidden'; document.body.appendChild(link); link.click(); document.body.removeChild(link); showToast('Export réussi !', 'success'); } function initPage() { renderDeliveries(); updateStats(); const role = localStorage.getItem('newketRole'); document.querySelectorAll('.admin-only').forEach(el => { if (role === 'admin') el.classList.remove('hidden'); else el.classList.add('hidden'); }); } document.addEventListener('DOMContentLoaded', () => { if (window.newketInitialized) { initPage(); } else { window.addEventListener('newketInitialized', initPage); } });
